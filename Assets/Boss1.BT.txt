tree("Root")
    fallback
        tree("CheckDead")
        tree("BeAlive")

tree("CheckDead")
    sequence
        not IsAlive
        Die

tree("BeAlive")
    fallback
        tree("TargetVisible")
        tree("Roam")
        
tree("TargetVisible")
    sequence
        IsTargetSeen
        mute
            fallback
                tree("StepOnTarget")
                tree("StepTowardsTarget")
        SwapCurrentFoot
        
tree("StepOnTarget")
    sequence
        TargetInStepRange
        SetFootDestinationOnTarget
        MoveFootToDestination
        Wait(0.2)

tree("StepTowardsTarget")
    sequence
        SetFootDestinationTowardsTarget
        MoveFootToDestination
        
        

tree("Roam")
    Succeed()
