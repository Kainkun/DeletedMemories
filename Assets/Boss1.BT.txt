tree("Root")
    fallback
        tree("CheckDead")
        tree("BeAlive")

tree("CheckDead")
    sequence
        not IsAlive
        Die

tree("BeAlive")
    fallback
        tree("TargetVisible")
        tree("Roam")
        
tree("TargetVisible")
    sequence
        IsTargetSeen
        fallback
            //tree("StepOnTarget")
            tree("StepTowardsTarget")
        
tree("StepOnTarget")
    sequence
        TargetInStepRange
        SetFootDestinationTowardsTarget
        MoveFootToDestination
        MoveFootToGround
        SwapCurrentFoot
        Wait(0.2)

tree("StepTowardsTarget")
    sequence
        SetFootDestinationTowardsTarget
        MoveFootToDestination
        //MoveFootToGround
        SwapCurrentFoot

tree("Roam")
    Succeed()
